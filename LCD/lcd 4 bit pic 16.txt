void lcd_cmd(unsigned char ch)
{
unsigned char temp;
temp=ch&0XF0;
LCD&=0X0F;
LCD|=temp;
RS=0;
EN=1;
delay(2);
EN=0;
temp=(ch&0X0F)<<4;
LCD&=0X0F;
LCD|=temp;
EN=1;
delay(2);
EN=0;
}
void lcd_data(unsigned char ch)
{
	unsigned char temp;
	temp=ch&0X0F;
	LCD&=0X0F;
	LCD|=temp;
	RS=1;
	EN=1;
delay(2);
EN=0;
temp=(ch&0X0F)<<4;
LCD&=0X0F;
LCD|=temp;
RS=1;
EN=1;
delay(2);
EN=0;
}
void lcd_init()
{
	lcd_cmd(0X02);
	lcd_cmd(0X28);
	lcd_cmd(0x06);
	lcd_cmd(0x01);
	}
	void lcd_print(unsigned char *str)
	{
	while(*str)
