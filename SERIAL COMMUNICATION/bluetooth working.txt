#include<htc.h>
void uart_init();
void uart_tx(char ch);
char uart_rx();
void main()
{
unsigned char m;
TRISD1=0;
TRISC6=0;
TRISC7=1;
	uart_init();
	uart_tx('c');
	uart_tx('o');
	uart_tx('n');
	uart_tx('n');
	uart_tx('e');
	uart_tx('c');
	uart_tx('t');
	uart_tx('e');
	uart_tx('d');
while(1)
{
	m=uart_rx();
	if(m=='1')
	{
		RD1=1;
	}
	if(m=='0')
	{
		RD1=0;
	}
}
}
void uart_init()
{
TXSTA=0X24;
RCSTA=0X90;
BRGH=1;
SPBRG=129;
}
void uart_tx(char ch)
{
	
	TXREG=ch;
	while(TXIF==0);
	{
		TXIF=0;
	}
}
char uart_rx()
{
	
	while(RCIF==0);
	RCIF=0;
	return RCREG;
}
